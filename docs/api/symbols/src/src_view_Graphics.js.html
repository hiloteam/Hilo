 53     /**
 54      * 指定绘制图形的线条样式。
 55      * @param {Number} thickness 线条的粗细值。默认为1。
 56      * @param {String} lineColor 线条的CSS颜色值。默认为黑色，即'0'。
 57      * @param {Number} lineAlpha 线条的透明度值。默认为不透明，即1。
 58      * @param {String} lineCap 线条的端部样式。可选值有：butt、round、square等，默认值为null。
 59      * @param {String} lineJoin 线条的连接样式。可选值有：miter、round、bevel等，默认值为null。
 60      * @param {Number} miterLimit 斜连线长度和线条宽度的最大比率。此属性仅当lineJoin为miter时有效。默认值为10。
 61      * @returns {Graphics} Graphics对象本身。

 71         if(miterLimit != undefined) addAction.call(me, ['miterLimit', (me.miterLimit = miterLimit)]);

109      * @param {Number} y1 渐变的结束点的y轴坐标。
110      * @param {Array} colors 渐变中使用的CSS颜色值数组。
111      * @param {Array} ratois 渐变中开始与结束之间的位置数组。需与colors数组里的颜色值一一对应，介于0.0与1.0之间的值。

232     /**
233      * 绘制一个复杂的圆角矩形。
234      * @param {Number} x x轴坐标。
235      * @param {Number} y y轴坐标。
236      * @param {Number} width 圆角矩形的宽度。
237      * @param {Number} height 圆角矩形的高度。

244     drawRoundRectComplex: function(x, y, width, height, cornerTL, cornerTR, cornerBR, cornerBL){
245         var me = this, addAction = me._addAction;

251         addAction.call(me, ['lineTo', x + cornerBL, y + height]);
252         addAction.call(me, ['arc', x + cornerBL, y + height - cornerBL, cornerBL, Math.PI/2, Math.PI, false]);

282     /**
283      * 绘制一个椭圆。
284      * @param {Number} x x轴坐标。
285      * @param {Number} y y轴坐标。
286      * @param {Number} width 椭圆的宽度。
287      * @param {Number} height 椭圆的高度。
288      * @returns {Graphics} Graphics对象本身。
289      */
290     drawEllipse: function(x, y, width, height){
291         var me = this;
292         if(width == height) return me.drawCircle(x, y, width);
293 
294         var addAction = me._addAction;
295         var w = width / 2, h = height / 2, C = 0.5522847498307933, cx = C * w, cy = C * h;
296         x = x + w;
297         y = y + h;
298 
299         addAction.call(me, ['moveTo', x + w, y]);
300         addAction.call(me, ['bezierCurveTo', x + w, y - cy, x + cx, y - h, x, y - h]);
301         addAction.call(me, ['bezierCurveTo', x - cx, y - h, x - w, y - cy, x - w, y]);
302         addAction.call(me, ['bezierCurveTo', x - w, y + cy, x - cx, y + h, x, y + h]);
303         addAction.call(me, ['bezierCurveTo', x + cx, y + h, x + w, y + cy, x + w, y]);
304         return me;
305     },

307     /**
308      * 根据参数指定的SVG数据绘制一条路径。
309      * 代码示例:
310      * <p>var path = 'M250 150 L150 350 L350 350 Z';</p>
311      * <p>var shape = new Hilo.Graphics({width:500, height:500});</p>
312      * <p>shape.drawSVGPath(path).beginFill('#0ff').endFill();</p>
313      * @param {String} pathData 要绘制的SVG路径数据。
314      * @returns {Graphics} Graphics对象本身。
315      */
316     drawSVGPath: function(pathData){
317         var me = this, addAction = me._addAction,
318             path = pathData.split(/,| (?=[a-zA-Z])/);
319 
320         addAction.call(me, ['beginPath']);
321         for(var i = 0, len = path.length; i < len; i++){
322             var str = path[i], cmd = str[0].toUpperCase(), p = str.substring(1).split(/,| /);
323             if(p[0].length == 0) p.shift();
324 
325             switch(cmd){
326                 case 'M':
327                     addAction.call(me, ['moveTo', p[0], p[1]]);
328                     break;
329                 case 'L':
330                     addAction.call(me, ['lineTo', p[0], p[1]]);
331                     break;
332                 case 'C':
333                     addAction.call(me, ['bezierCurveTo', p[0], p[1], p[2], p[3], p[4], p[5]]);
334                     break;
335                 case 'Z':
336                     addAction.call(me, ['closePath']);
337                     break;
338             }
339         }
340         return me;
341     },

343     /**
344      * 执行全部绘制动作。内部私有方法。
345      * @private
346      */
347     _draw: function(context){
348         var me = this, actions = me._actions, len = actions.length, i;
349 
350         context.beginPath();
351         for(i = 0; i < len; i++){
352             var action = actions[i],
353                 f = action[0],
354                 args = action.length > 1 ? action.slice(1) : null;
355 
356             if(typeof(context[f]) == 'function') context[f].apply(context, args);
357             else context[f] = action[1];
358         }
359     },

366         var me = this, canvas = renderer.canvas;
367         if(renderer.renderType === 'canvas'){
368             me._draw(renderer.context);
369         }else{
370             me.cache();
371             renderer.draw(me);
372         }
