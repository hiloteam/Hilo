  7 /**
  8  * @class BitmapText类提供使用位图文本的功能。当前仅支持单行文本。
  9  * @augments Container
 10  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
 11  * @module hilo/view/BitmapText
 12  * @requires hilo/core/Class
 13  * @requires hilo/core/Hilo
 14  * @requires hilo/view/Container
 15  * @requires hilo/view/Bitmap
 16  * @property {Object} glyphs 位图字体的字形集合。格式为：{letter:{image:img, rect:[0,0,100,100]}}。
 17  * @property {Number} letterSpacing 字距，即字符间的间隔。默认值为0。
 18  * @property {String} text 位图文本的文本内容。只读属性。设置文本请使用setFont方法。
 19  * @property {String} textAlign 文本对齐方式，值为left、center、right, 默认left。只读属性。设置文本请使用setTextAlign方法。
 20  */
 21 var BitmapText = Class.create(/** @lends BitmapText.prototype */{
 22     Extends: Container,
 23     constructor: function(properties){
 24         properties = properties || {};
 25         this.id = this.id || properties.id || Hilo.getUid('BitmapText');
 26         BitmapText.superclass.constructor.call(this, properties);
 27 
 28         var text = properties.text + '';
 29         if(text){
 30             this.text = '';
 31             this.setText(text);
 32         }
 33 
 34         this.pointerChildren = false; //disable user events for single letters
 35     },
 36 
 37     glyphs: null,
 38     letterSpacing: 0,
 39     text: '',
 40     textAlign:'left',
 41 
 42     /**
 43      * 设置位图文本的文本内容。
 44      * @param {String} text 要设置的文本内容。
 45      * @returns {BitmapText} BitmapText对象本身。链式调用支持。
 46      */
 47     setText: function(text){
 48         var me = this, str = text.toString(), len = str.length;
 49         if(me.text == str) return;
 50         me.text = str;
 51 
 52         var i, charStr, charGlyph, charObj, width = 0, height = 0, left = 0;
 53         for(i = 0; i < len; i++){
 54             charStr = str.charAt(i);
 55             charGlyph = me.glyphs[charStr];
 56             if(charGlyph){
 57                 left = width + (width > 0 ? me.letterSpacing : 0);
 58                 if(me.children[i]){
 59                     charObj = me.children[i];
 60                     charObj.setImage(charGlyph.image, charGlyph.rect);
 61                 }
 62                 else{
 63                     charObj = me._createBitmap(charGlyph);
 64                     me.addChild(charObj);
 65                 }
 66                 charObj.x = left;
 67                 width = left + charGlyph.rect[2];
 68                 height = Math.max(height, charGlyph.rect[3]);
 69             }
 70         }
 71 
 72         for(i = me.children.length - 1;i >= len;i --){
 73             me._releaseBitmap(me.children[i]);
 74             me.children[i].removeFromParent();
 75         }
 76 
 77         me.width = width;
 78         me.height = height;
 79         this.setTextAlign();
 80         return me;
 81     },
 82     _createBitmap:function(cfg){
 83         var bmp;
 84         if(BitmapText._pool.length){
 85             bmp = BitmapText._pool.pop();
 86             bmp.setImage(cfg.image, cfg.rect);
 87         }
 88         else{
 89             bmp = new Bitmap({
 90                 image:cfg.image,
 91                 rect:cfg.rect
 92             });
 93         }
 94         return bmp;
 95     },
 96     _releaseBitmap:function(bmp){
 97         BitmapText._pool.push(bmp);
 98     },
 99 
100      /**
101      * 设置位图文本的对齐方式。
102      * @param textAlign 文本对齐方式，值为left、center、right
103      * @returns {BitmapText} BitmapText对象本身。链式调用支持。
104      */
105     setTextAlign:function(textAlign){
106         this.textAlign = textAlign||this.textAlign;
107         switch(this.textAlign){
108             case "center":
109                 this.pivotX = this.width * .5;
110                 break;
111             case "right":
112                 this.pivotX = this.width;
113                 break;
114             case "left":
115             default:
116                 this.pivotX = 0;
117                 break;
118         }
119         return this;
120     },
121 
122     /**
123      * 返回能否使用当前指定的字体显示提供的字符串。
124      * @param {String} str 要检测的字符串。
125      * @returns {Boolean} 是否能使用指定字体。
126      */
127     hasGlyphs: function(str){
128         var glyphs = this.glyphs;
129         if(!glyphs) return false;
130 
131         var str = str.toString(), len = str.length, i;
132         for(i = 0; i < len; i++){
133             if(!glyphs[str.charAt(i)]) return false;
134         }
135         return true;
136     },
137 
138     Statics:/** @lends BitmapText */{
139         _pool:[],
140         /**
141          * 简易方式生成字形集合。
142          * @static
143          * @param {String} text 字符文本。
144          * @param {Image} image 字符图片。
145          * @param {Number} col 列数  默认和文本字数一样
146          * @param {Number} row 行数 默认1行
147          * @returns {BitmapText} BitmapText对象本身。链式调用支持。
148          */
149         createGlyphs:function(text, image, col, row){
150             var str = text.toString();
151             col = col||str.length;
152             row = row||1;
153             var w = image.width/col;
154             var h = image.height/row;
155             var glyphs = {};
156             for(var i = 0, l = text.length;i < l;i ++){
157                 charStr = str.charAt(i);
158                 glyphs[charStr] = {
159                     image:image,
160                     rect:[w * (i % col), h * Math.floor(i / col), w, h]
161                 }
162             }
163             return glyphs;
164         }
165     }
166 
167 });
