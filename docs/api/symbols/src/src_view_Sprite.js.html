  7 /**
  8  * @class 动画精灵类。
  9  * @augments View
 10  * @module hilo/view/Sprite
 11  * @requires hilo/core/Hilo
 12  * @requires hilo/core/Class
 13  * @requires hilo/view/View
 14  * @requires hilo/view/Drawable
 15  * @param properties 创建对象的属性参数。可包含此类所有可写属性。此外还包括：
 16  * <ul>
 17  * <li><b>frames</b> - 精灵动画的帧数据对象。</li>
 18  * </ul>
 19  * @property {number} currentFrame 当前播放帧的索引。从0开始。只读属性。
 20  * @property {boolean} paused 判断精灵是否暂停。默认为false。
 21  * @property {boolean} loop 判断精灵是否可以循环播放。默认为true。
 22  * @property {boolean} timeBased 指定精灵动画是否是以时间为基准。默认为false，即以帧为基准。
 23  * @property {number} interval 精灵动画的帧间隔。如果timeBased为true，则单位为毫秒，否则为帧数。
 24  */
 25 var Sprite = Class.create(/** @lends Sprite.prototype */{
 26     Extends: View,
 27     constructor: function(properties){
 28         properties = properties || {};
 29         this.id = this.id || properties.id || Hilo.getUid("Sprite");
 30         Sprite.superclass.constructor.call(this, properties);
 31 
 32         this._frames = [];
 33         this._frameNames = {};
 34         this.drawable = new Drawable();
 35         if(properties.frames) this.addFrame(properties.frames);
 36     },
 37 
 38     _frames: null, //所有帧的集合
 39     _frameNames: null, //带名字name的帧的集合
 40     _frameElapsed: 0, //当前帧持续的时间或帧数
 41     _firstRender: true, //标记是否是第一次渲染
 42 
 43     paused: false,
 44     loop: true,
 45     timeBased: false,
 46     interval: 1,
 47     currentFrame: 0, //当前帧的索引
 48 
 49     /**
 50      * 返回精灵动画的总帧数。
 51      * @returns {Uint} 精灵动画的总帧数。
 52      */
 53     getNumFrames: function(){
 54         return this._frames ? this._frames.length : 0;
 55     },
 56 
 57     /**
 58      * 往精灵动画序列中增加帧。
 59      * @param {Object} frame 要增加的精灵动画帧数据。
 60      * @param {Int} startIndex 开始增加帧的索引位置。若不设置，默认为在末尾添加。
 61      * @returns {Sprite} Sprite对象本身。
 62      */
 63     addFrame: function(frame, startIndex){
 64         var start = startIndex != null ? startIndex : this._frames.length;
 65         if(frame instanceof Array){
 66             for(var i = 0, len = frame.length; i < len; i++){
 67                 this.setFrame(frame[i], start + i);
 68             }
 69         }else{
 70             this.setFrame(frame, start);
 71         }
 72         return this;
 73     },
 74 
 75     /**
 76      * 设置精灵动画序列指定索引位置的帧。
 77      * @param {Object} frame 要设置的精灵动画帧数据。
 78      * @param {Int} index 要设置的索引位置。
 79      * @returns {Sprite} Sprite对象本身。
 80      */
 81     setFrame: function(frame, index){
 82         var frames = this._frames,
 83             total = frames.length;
 84         index = index < 0 ? 0 : index > total ? total : index;
 85         frames[index] = frame;
 86         if(frame.name) this._frameNames[frame.name] = frame;
 87         if(index == 0 && !this.width || !this.height){
 88             this.width = frame.rect[2];
 89             this.height = frame.rect[3];
 90         }
 91         return this;
 92     },
 93 
 94     /**
 95      * 获取精灵动画序列中指定的帧。
 96      * @param {Object} indexOrName 要获取的帧的索引位置或别名。
 97      * @returns {Object} 精灵帧对象。
 98      */
 99     getFrame: function(indexOrName){
100         if(typeof indexOrName === 'number'){
101             var frames = this._frames;
102             if(indexOrName < 0 || indexOrName >= frames.length) return null;
103             return frames[indexOrName];
104         }
105         return this._frameNames[indexOrName];
106     },
107 
108     /**
109      * 获取精灵动画序列中指定帧的索引位置。
110      * @param {Object} frameValue 要获取的帧的索引位置或别名。
111      * @returns {Object} 精灵帧对象。
112      */
113     getFrameIndex: function(frameValue){
114         var frames = this._frames,
115             total = frames.length,
116             index = -1;
117         if(typeof frameValue === 'number'){
118             index = frameValue;
119         }else{
120             var frame = typeof frameValue === 'string' ? this._frameNames[frameValue] : frameValue;
121             if(frame){
122                 for(var i = 0; i < total; i++){
123                     if(frame === frames[i]){
124                         index = i;
125                         break;
126                     }
127                 }
128             }
129         }
130         return index;
131     },
132 
133     /**
134      * 播放精灵动画。
135      * @returns {Sprite} Sprite对象本身。
136      */
137     play: function(){
138         this.paused = false;
139         return this;
140     },
141 
142     /**
143      * 暂停播放精灵动画。
144      * @returns {Sprite} Sprite对象本身。
145      */
146     stop: function(){
147         this.paused = true;
148         return this;
149     },
150 
151     /**
152      * 跳转精灵动画到指定的帧。
153      * @param {Object} indexOrName 要跳转的帧的索引位置或别名。
154      * @param {Boolean} pause 指示跳转后是否暂停播放。
155      * @returns {Sprite} Sprite对象本身。
156      */
157     goto: function(indexOrName, pause){
158         var total = this._frames.length,
159             index = this.getFrameIndex(indexOrName);
160 
161         this.currentFrame = index < 0 ? 0 : index >= total ? total - 1 : index;
162         this.paused = pause;
163         this._firstRender = true;
164         return this;
165     },
166 
167     /**
168      * 渲染方法。
169      * @private
170      */
171     _render: function(renderer, delta){
172         var lastFrameIndex = this.currentFrame, frameIndex;
173 
174         if(this._firstRender){
175             frameIndex = lastFrameIndex;
176             this._firstRender = false;
177         }else{
178             frameIndex = this._nextFrame(delta);
179         }
180 
181         if(frameIndex != lastFrameIndex){
182             this.currentFrame = frameIndex;
183             var callback = this._frames[frameIndex].callback;
184             callback && callback.call(this);
185         }
186 
187         //NOTE: it will be deprecated, don't use it.
188         if(this.onEnterFrame) this.onEnterFrame(frameIndex);
189 
190         this.drawable.init(this._frames[frameIndex]);
191         Sprite.superclass._render.call(this, renderer, delta);
192     },
193 
194     /**
195      * @private
196      */
197     _nextFrame: function(delta){
198         var frames = this._frames,
199             total = frames.length,
200             frameIndex = this.currentFrame,
201             frame = frames[frameIndex],
202             duration = frame.duration || this.interval,
203             elapsed = this._frameElapsed;
204 
205         //calculate the current frame elapsed frames/time
206         var value = (frameIndex == 0 && !this.drawable) ? 0 : elapsed + (this.timeBased ? delta : 1);
207         elapsed = this._frameElapsed = value < duration ? value : 0;
208 
209         if(frame.stop || !this.loop && frameIndex >= total - 1){
210             this.stop();
211         }
212 
213         if(!this.paused && elapsed == 0){
214             if(frame.next != null){
215                 //jump to the specified frame
216                 frameIndex = this.getFrameIndex(frame.next);
217             }else if(frameIndex >= total - 1){
218                 //at the end of the frames, go back to first frame
219                 frameIndex = 0;
220             }else if(this.drawable){
221                 //normal go forward to next frame
222                 frameIndex++;
223             }
224         }
225 
226         return frameIndex;
227     },
228 
229     /**
230      * 设置指定帧的回调函数。即每当播放头进入指定帧时调用callback函数。若callback为空，则会删除回调函数。
231      * @param {Int|String} frame 要指定的帧的索引位置或别名。
232      * @param {Function} callback 指定回调函数。
233      * @returns {Sprite} 精灵本身。
234      */
235     setFrameCallback: function(frame, callback){
236         frame = this.getFrame(frame);
237         if(frame) frame.callback = callback;
238         return this;
239     },
240 
241     /**
242      * 精灵动画的播放头进入新帧时的回调方法。默认值为null。此方法已废弃，请使用addFrameCallback方法。
243      * @type Function
244      * @deprecated
245      */
246     onEnterFrame: null
247 
248 });
