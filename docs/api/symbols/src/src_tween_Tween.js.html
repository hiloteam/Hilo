  7 /**
  8  * @class Tween类提供缓动功能。
  9  * @param {Object} target 缓动对象。
 10  * @param {Object} fromProps 对象缓动的起始属性集合。
 11  * @param {Object} toProps 对象缓动的目标属性集合。
 12  * @param {Object} params 缓动参数。可包含Tween类所有可写属性。
 13  * @module hilo/tween/Tween
 14  * @requires hilo/core/Class
 15  * @property {Object} target 缓动目标。只读属性。
 16  * @property {Int} duration 缓动总时长。单位毫秒。
 17  * @property {Int} delay 缓动延迟时间。单位毫秒。
 18  * @property {Boolean} paused 缓动是否暂停。默认为false。
 19  * @property {Boolean} loop 缓动是否循环。默认为false。
 20  * @property {Boolean} reverse 缓动是否反转播放。默认为false。
 21  * @property {Int} repeat 缓动重复的次数。默认为0。
 22  * @property {Int} repeatDelay 缓动重复的延迟时长。单位为毫秒。
 23  * @property {Function} ease 缓动变化函数。默认为null。
 24  * @property {Int} time 缓动已进行的时长。单位毫秒。只读属性。
 25  * @property {Function} onStart 缓动开始回调函数。它接受1个参数：tween。默认值为null。
 26  * @property {Function} onUpdate 缓动更新回调函数。它接受2个参数：ratio和tween。默认值为null。
 27  * @property {Function} onComplete 缓动结束回调函数。它接受1个参数：tween。默认值为null。
 28  */
 29 var Tween = (function(){
 30 
 31 function now(){
 32     return +new Date();
 33 }
 34 
 35 return Class.create(/** @lends Tween.prototype */{
 36     constructor: function(target, fromProps, toProps, params){
 37         var me = this;
 38 
 39         me.target = target;
 40         me._startTime = 0;
 41         me._seekTime = 0;
 42         me._pausedTime = 0;
 43         me._pausedStartTime = 0;
 44         me._reverseFlag = 1;
 45         me._repeatCount = 0;
 46 
 47         //no fromProps if pass 3 arguments
 48         if(arguments.length == 3){
 49             params = toProps;
 50             toProps = fromProps;
 51             fromProps = null;
 52         }
 53 
 54         for(var p in params) me[p] = params[p];
 55         me.setProps(fromProps, toProps);
 56 
 57         //for old version compatiblity
 58         if(!params.duration && params.time){
 59             me.duration = params.time || 0;
 60             me.time = 0;
 61         }
 62     },
 63 
 64     target: null,
 65     duration: 0,
 66     delay: 0,
 67     paused: false,
 68     loop: false,
 69     reverse: false,
 70     repeat: 0,
 71     repeatDelay: 0,
 72     ease: null,
 73     time: 0, //ready only
 74 
 75     onStart: null,
 76     onUpdate: null,
 77     onComplete: null,
 78 
 79     /**
 80      * 设置缓动对象的初始和目标属性。
 81      * @param {Object} fromProps 缓动对象的初始属性。
 82      * @param {Object} toProps 缓动对象的目标属性。
 83      * @returns {Tween} Tween变换本身。可用于链式调用。
 84      */
 85     setProps: function(fromProps, toProps){
 86         var me = this, target = me.target,
 87             propNames = fromProps || toProps,
 88             from = me._fromProps = {}, to = me._toProps = {};
 89 
 90         fromProps = fromProps || target;
 91         toProps = toProps || target;
 92 
 93         for(var p in propNames){
 94             to[p] = toProps[p] || 0;
 95             target[p] = from[p] = fromProps[p] || 0;
 96         }
 97         return me;
 98     },
 99 
100     /**
101      * 启动缓动动画的播放。
102      * @returns {Tween} Tween变换本身。可用于链式调用。
103      */
104     start: function(){
105         var me = this;
106         me._startTime = now() + me.delay;
107         me._seekTime = 0;
108         me._pausedTime = 0;
109         me.paused = false;
110         Tween.add(me);
111         return me;
112     },
113 
114     /**
115      * 停止缓动动画的播放。
116      * @returns {Tween} Tween变换本身。可用于链式调用。
117      */
118     stop: function(){
119         Tween.remove(this);
120         return this;
121     },
122 
123     /**
124      * 暂停缓动动画的播放。
125      * @returns {Tween} Tween变换本身。可用于链式调用。
126      */
127     pause: function(){
128         var me = this;
129         me.paused = true;
130         me._pausedStartTime = now();
131         return me;
132     },
133 
134     /**
135      * 恢复缓动动画的播放。
136      * @returns {Tween} Tween变换本身。可用于链式调用。
137      */
138     resume: function(){
139         var me = this;
140         me.paused = false;
141         if(me._pausedStartTime) me._pausedTime += now() - me._pausedStartTime;
142         me._pausedStartTime = 0;
143         return me;
144     },
145 
146     /**
147      * 跳转Tween到指定的时间。
148      * @param {Number} time 指定要跳转的时间。取值范围为：0 - duraion。
149      * @param {Boolean} pause 是否暂停。
150      * @returns {Tween} Tween变换本身。可用于链式调用。
151      */
152     seek: function(time, pause){
153         var me = this, current = now();
154         me._startTime = current;
155         me._seekTime = time;
156         me._pausedTime = 0;
157         if(pause !== undefined) me.paused = pause;
158         me._update(current, true);
159         Tween.add(me);
160         return me;
161     },
162 
163     /**
164      * 连接下一个Tween变换。其开始时间根据delay值不同而不同。当delay值为字符串且以'+'或'-'开始时，Tween的开始时间从当前变换结束点计算，否则以当前变换起始点计算。
165      * @param {Tween} tween 要连接的Tween变换。
166      * @returns {Tween} Tween变换本身。可用于链式调用。
167      */
168     link: function(tween){
169         var me = this, delay = tween.delay, startTime = me._startTime;
170 
171         if(typeof delay === 'string'){
172             var plus = delay.indexOf('+') == 0, minus = delay.indexOf('-') == 0;
173             delay = plus || minus ? Number(delay.substr(1)) * (plus ? 1 : -1) : Number(delay);
174         }
175         tween.delay = delay;
176         tween._startTime = plus || minus ? startTime + me.duration + delay : startTime + delay;
177 
178         me._next = tween;
179         Tween.remove(tween);
180         return me;
181     },
182 
183     /**
184      * Tween类的内部渲染方法。
185      * @private
186      */
187     _render: function(ratio){
188         var me = this, target = me.target, fromProps = me._fromProps, p;
189         for(p in fromProps) target[p] = fromProps[p] + (me._toProps[p] - fromProps[p]) * ratio;
190     },
191 
192     /**
193      * Tween类的内部更新方法。
194      * @private
195      */
196     _update: function(time, forceUpdate){
197         var me = this;
198         if(me.paused && !forceUpdate) return;
199 
200         //elapsed time
201         var elapsed = time - me._startTime - me._pausedTime + me._seekTime;
202         if(elapsed < 0) return;
203 
204         //elapsed ratio
205         var ratio = elapsed / me.duration, complete = false, callback;
206         ratio = ratio <= 0 ? 0 : ratio >= 1 ? 1 : me.ease ? me.ease(ratio) : ratio;
207 
208         if(me.reverse){
209             //backward
210             if(me._reverseFlag < 0) ratio = 1 - ratio;
211             //forward
212             if(ratio < 1e-7){
213                 //repeat complete or not loop
214                 if((me.repeat > 0 && me._repeatCount++ >= me.repeat) || (me.repeat == 0 && !me.loop)){
215                     complete = true;
216                 }else{
217                     me._startTime = now();
218                     me._pausedTime = 0;
219                     me._reverseFlag *= -1;
220                 }
221             }
222         }
223 
224         //start callback
225         if(me.time == 0 && (callback = me.onStart)) callback.call(me, me);
226         me.time = elapsed;
227 
228         //render & update callback
229         me._render(ratio);
230         (callback = me.onUpdate) && callback.call(me, ratio, me);
231 
232         //check if complete
233         if(ratio >= 1){
234             if(me.reverse){
235                 me._startTime = now();
236                 me._pausedTime = 0;
237                 me._reverseFlag *= -1;
238             }else if(me.loop || me.repeat > 0 && me._repeatCount++ < me.repeat){
239                 me._startTime = now() + me.repeatDelay;
240                 me._pausedTime = 0;
241             }else{
242                 complete = true;
243             }
244         }
245 
246         //next tween
247         var next = me._next;
248         if(next && next.time <= 0){
249             var nextStartTime = next._startTime;
250             if(nextStartTime > 0 && nextStartTime <= time){
251                 //parallel tween
252                 next._render(ratio);
253                 next.time = elapsed;
254                 Tween.add(next);
255             }else if(complete && (nextStartTime < 0 || nextStartTime > time)){
256                 //next tween
257                 next.start();
258             }
259         }
260 
261         //complete
262         if(complete){
263             (callback = me.onComplete) && callback.call(me, me);
264             return true;
265         }
266     },
267 
268     Statics: /** @lends Tween */ {
269         /**
270          * @private
271          */
272         _tweens: [],
273 
274         /**
275          * 更新所有Tween实例。
276          * @returns {Object} Tween。
277          */
278         tick: function(){
279             var tweens = Tween._tweens, tween, i, len = tweens.length;
280 
281             for(i = 0; i < len; i++){
282                 tween = tweens[i];
283                 if(tween && tween._update(now())){
284                     tweens.splice(i, 1);
285                     i--;
286                 }
287             }
288             return Tween;
289         },
290 
291         /**
292          * 添加Tween实例。
293          * @param {Tween} tween 要添加的Tween对象。
294          * @returns {Object} Tween。
295          */
296         add: function(tween){
297             var tweens = Tween._tweens;
298             if(tweens.indexOf(tween) == -1) tweens.push(tween);
299             return Tween;
300         },
301 
302         /**
303          * 删除Tween实例。
304          * @param {Tween|Object|Array} tweenOrTarget 要删除的Tween对象或target对象或要删除的一组对象。
305          * @returns {Object} Tween。
306          */
307         remove: function(tweenOrTarget){
308             if(tweenOrTarget instanceof Array){
309                 for(var i = 0, l = tweenOrTarget.length;i < l;i ++){
310                     Tween.remove(tweenOrTarget[i]);
311                 }
312                 return Tween;
313             }
314 
315             var tweens = Tween._tweens, i;
316             if(tweenOrTarget instanceof Tween){
317                 i = tweens.indexOf(tweenOrTarget);
318                 if(i > -1) tweens.splice(i, 1);
319             }else{
320                 for(i = 0; i < tweens.length; i++){
321                     if(tweens[i].target === tweenOrTarget){
322                         tweens.splice(i, 1);
323                         i--;
324                     }
325                 }
326             }
327 
328             return Tween;
329         },
330 
331         /**
332          * 删除所有Tween实例。
333          * @returns {Object} Tween。
334          */
335         removeAll: function(){
336             Tween._tweens.length = 0;
337             return Tween;
338         },
339 
340         /**
341          * 创建一个缓动动画，让目标对象从开始属性变换到目标属性。
342          * @param {Object|Array} target 缓动目标对象或缓动目标数组。
343          * @param fromProps 缓动目标对象的开始属性。
344          * @param toProps 缓动目标对象的目标属性。
345          * @param params 缓动动画的参数。
346          * @returns {Tween|Array} 一个Tween实例对象或Tween实例数组。
347          */
348         fromTo: function(target, fromProps, toProps, params){
349             var isArray = target instanceof Array;
350             target = isArray ? target : [target];
351 
352             var tween, i, stagger = params.stagger, tweens = [];
353             for(i = 0; i < target.length; i++){
354                 tween = new Tween(target[i], fromProps, toProps, params);
355                 if(stagger) tween.delay = (params.delay || 0) + (i * stagger || 0);
356                 tween.start();
357                 tweens.push(tween);
358             }
359 
360             return isArray?tweens:tween;
361         },
362 
363         /**
364          * 创建一个缓动动画，让目标对象从当前属性变换到目标属性。
365          * @param {Object|Array} target 缓动目标对象或缓动目标数组。
366          * @param toProps 缓动目标对象的目标属性。
367          * @param params 缓动动画的参数。
368          * @returns {Tween|Array} 一个Tween实例对象或Tween实例数组。
369          */
370         to: function(target, toProps, params){
371             return Tween.fromTo(target, null, toProps, params);
372         },
373 
374         /**
375          * 创建一个缓动动画，让目标对象从指定的起始属性变换到当前属性。
376          * @param {Object|Array} target 缓动目标对象或缓动目标数组。
377          * @param fromProps 缓动目标对象的目标属性。
378          * @param params 缓动动画的参数。
379          * @returns {Tween|Array} 一个Tween实例对象或Tween实例数组。
380          */
381         from: function(target, fromProps, params){
382             return Tween.fromTo(target, fromProps, null, params);
383         }
384     }
385 
386 });
387 
388 })();
