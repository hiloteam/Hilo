  7 /**
  8  * 示例:
  9  * <pre>
 10  * var btn = new Hilo.Button({
 11  *     image: buttonImage,
 12  *     upState: {rect:[0, 0, 64, 64]},
 13  *     overState: {rect:[64, 0, 64, 64]},
 14  *     downState: {rect:[128, 0, 64, 64]},
 15  *     disabledState: {rect:[192, 0, 64, 64]}
 16  * });
 17  * </pre>
 18  * @class Button类表示简单按钮类。它有弹起、经过、按下和不可用等四种状态。
 19  * @augments View
 20  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。此外还包括：
 21  * <ul>
 22  * <li><b>image</b> - 按钮图片所在的image对象。</li>
 23  * </ul>
 24  * @module hilo/view/Button
 25  * @requires hilo/core/Hilo
 26  * @requires hilo/core/Class
 27  * @requires hilo/view/View
 28  * @requires hilo/view/Drawable
 29  * @property {Object} upState 按钮弹起状态的属性或其drawable的属性的集合。
 30  * @property {Object} overState 按钮经过状态的属性或其drawable的属性的集合。
 31  * @property {Object} downState 按钮按下状态的属性或其drawable的属性的集合。
 32  * @property {Object} disabledState 按钮不可用状态的属性或其drawable的属性的集合。
 33  * @property {String} state 按钮的状态名称。它是 Button.UP|OVER|DOWN|DISABLED 之一。 只读属性。
 34  * @property {Boolean} enabled 指示按钮是否可用。默认为true。只读属性。
 35  * @property {Boolean} useHandCursor 当设置为true时，表示指针滑过按钮上方时是否显示手形光标。默认为true。
 36  */
 37  var Button = Class.create(/** @lends Button.prototype */{
 38     Extends: View,
 39     constructor: function(properties){
 40         properties = properties || {};
 41         this.id = this.id || properties.id || Hilo.getUid("Button");
 42         Button.superclass.constructor.call(this, properties);
 43 
 44         this.drawable = new Drawable(properties);
 45         this.setState(Button.UP);
 46     },
 47 
 48     upState: null,
 49     overState: null,
 50     downState: null,
 51     disabledState: null,
 52 
 53     state: null,
 54     enabled: true,
 55     useHandCursor: true,
 56 
 57     /**
 58      * 设置按钮是否可用。
 59      * @param {Boolean} enabled 指示按钮是否可用。
 60      * @returns {Button} 按钮本身。
 61      */
 62     setEnabled: function(enabled){
 63         if(this.enabled != enabled){
 64             if(!enabled){
 65                 this.setState(Button.DISABLED);
 66             }else{
 67                 this.setState(Button.UP);
 68             }
 69         }
 70         return this;
 71     },
 72 
 73     /**
 74      * 设置按钮的状态。此方法由Button内部调用，一般无需使用此方法。
 75      * @param {String} state 按钮的新的状态。
 76      * @returns {Button} 按钮本身。
 77      */
 78     setState: function(state){
 79         if(this.state !== state){
 80             this.state = state;
 81             this.pointerEnabled = this.enabled = state !== Button.DISABLED;
 82 
 83             var stateObj;
 84             switch(state){
 85                 case Button.UP:
 86                     stateObj = this.upState;
 87                     break;
 88                 case Button.OVER:
 89                     stateObj = this.overState;
 90                     break;
 91                 case Button.DOWN:
 92                     stateObj = this.downState;
 93                     break;
 94                 case Button.DISABLED:
 95                     stateObj = this.disabledState;
 96                     break;
 97             }
 98 
 99             if(stateObj){
100                 this.drawable.init(stateObj);
101                 Hilo.copy(this, stateObj, true);
102             }
103         }
104 
105         return this;
106     },
107 
108     /**
109      * overwrite
110      * @private
111      */
112     fire: function(type, detail){
113         if(!this.enabled) return;
114 
115         var evtType = typeof type === 'string' ? type : type.type;
116         switch(evtType){
117             case 'mousedown':
118             case 'touchstart':
119             case 'touchmove':
120                 this.setState(Button.DOWN);
121                 break;
122             case "mouseover":
123                 this.setState(Button.OVER);
124                 break;
125             case 'mouseup':
126                 if(this.overState) this.setState(Button.OVER);
127                 else if(this.upState) this.setState(Button.UP);
128                 break;
129             case 'touchend':
130             case 'touchout':
131             case 'mouseout':
132                 this.setState(Button.UP);
133                 break;
134         }
135 
136         return Button.superclass.fire.call(this, type, detail);
137     },
138 
139     Statics: /** @lends Button */ {
140         /**
141          * 按钮弹起状态的常量值，即：'up'。
142          * @type String
143          */
144         UP: 'up',
145         /**
146          * 按钮经过状态的常量值，即：'over'。
147          * @type String
148          */
149         OVER: 'over',
150         /**
151          * 按钮按下状态的常量值，即：'down'。
152          * @type String
153          */
154         DOWN: 'down',
155         /**
156          * 按钮不可用状态的常量值，即：'disabled'。
157          * @type String
158          */
159         DISABLED: 'disabled'
160     }
161  });
