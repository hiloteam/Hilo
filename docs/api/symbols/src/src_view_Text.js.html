  7 /**
  8  * @class Text类提供简单的文字显示功能。复杂的文本功能可以使用DOMElement。
  9  * @augments View
 10  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
 11  * @module hilo/view/Text
 12  * @requires hilo/core/Class
 13  * @requires hilo/core/Hilo
 14  * @requires hilo/view/View
 15  * @requires hilo/view/CacheMixin
 16  * @property {String} text 指定要显示的文本内容。
 17  * @property {String} color 指定使用的字体颜色。
 18  * @property {String} textAlign 指定文本的对齐方式。可以是以下任意一个值：'start', 'end', 'left', 'right', and 'center'。
 19  * @property {String} textVAlign 指定文本的垂直对齐方式。可以是以下任意一个值：'top', 'middle', 'bottom'。
 20  * @property {Boolean} outline 指定文本是绘制边框还是填充。
 21  * @property {Number} lineSpacing 指定文本的行距。单位为像素。默认值为0。
 22  * @property {Number} maxWidth 指定文本的最大宽度。默认值为200。
 23  * @property {String} font 文本的字体CSS样式。只读属性。设置字体样式请用setFont方法。
 24  * @property {Number} textWidth 指示文本内容的宽度，只读属性。仅在canvas模式下有效。
 25  * @property {Number} textHeight 指示文本内容的高度，只读属性。仅在canvas模式下有效。
 26  */
 27 var Text = Class.create(/** @lends Text.prototype */{
 28     Extends: View,
 29     Mixes:CacheMixin,
 30     constructor: function(properties){
 31         properties = properties || {};
 32         this.id = this.id || properties.id || Hilo.getUid('Text');
 33         Text.superclass.constructor.call(this, properties);
 34 
 35         // if(!properties.width) this.width = 200; //default width
 36         if(!properties.font) this.font = '12px arial'; //default font style
 37         this._fontHeight = Text.measureFontHeight(this.font);
 38     },
 39 
 40     text: null,
 41     color: '#000',
 42     textAlign: null,
 43     textVAlign: null,
 44     outline: false,
 45     lineSpacing: 0,
 46     maxWidth: 200,
 47     font: null, //ready-only
 48     textWidth: 0, //read-only
 49     textHeight: 0, //read-only
 50 
 51     /**
 52      * 设置文本的字体CSS样式。
 53      * @param {String} font 要设置的字体CSS样式。
 54      * @returns {Text} Text对象本身。链式调用支持。
 55      */
 56     setFont: function(font){
 57         var me = this;
 58         if(me.font !== font){
 59             me.font = font;
 60             me._fontHeight = Text.measureFontHeight(font);
 61         }
 62 
 63         return me;
 64     },
 65 
 66     /**
 67      * 覆盖渲染方法。
 68      * @private
 69      */
 70     render: function(renderer, delta){
 71         var me = this, canvas = renderer.canvas;
 72 
 73         if(renderer.renderType === 'canvas'){
 74             me._draw(renderer.context);
 75         }
 76         else if(renderer.renderType === 'dom'){
 77             var drawable = me.drawable;
 78             var domElement = drawable.domElement;
 79             var style = domElement.style;
 80 
 81             style.font = me.font;
 82             style.textAlign = me.textAlign;
 83             style.color = me.color;
 84             style.width = me.width + 'px';
 85             style.height = me.height + 'px';
 86             style.lineHeight = (me._fontHeight + me.lineSpacing) + 'px';
 87 
 88             domElement.innerHTML = me.text;
 89             renderer.draw(this);
 90         }
 91         else{
 92             //TODO:自动更新cache
 93             me.cache();
 94             renderer.draw(me);
 95         }
 96     },
 97 
 98     /**
 99      * 在指定的渲染上下文上绘制文本。
100      * @private
101      */
102     _draw: function(context){
103         var me = this, text = me.text.toString();
104         if(!text) return;
105 
106         //set drawing style
107         context.font = me.font;
108         context.textAlign = me.textAlign;
109         context.textBaseline = 'top';
110 
111         //find and draw all explicit lines
112         var lines = text.split(/\r\n|\r|\n|<br(?:[ \/])*>/);
113         var width = 0, height = 0;
114         var lineHeight = me._fontHeight + me.lineSpacing;
115         var i, line, w;
116         var drawLines = [];
117 
118         for(i = 0, len = lines.length; i < len; i++){
119             line = lines[i];
120             w = context.measureText(line).width;
121 
122             //check if the line need to split
123             if(w <= me.maxWidth){
124                 drawLines.push({text:line, y:height});
125                 // me._drawTextLine(context, line, height);
126                 if(width < w) width = w;
127                 height += lineHeight;
128                 continue;
129             }
130 
131             var str = '', oldWidth = 0, newWidth, j, word;
132 
133             for(j = 0, wlen = line.length; j < wlen; j++){
134                 word = line[j];
135                 newWidth = context.measureText(str + word).width;
136 
137                 if(newWidth > me.maxWidth){
138                     drawLines.push({text:str, y:height});
139                     // me._drawTextLine(context, str, height);
140                     if(width < oldWidth) width = oldWidth;
141                     height += lineHeight;
142                     str = word;
143                 }else{
144                     oldWidth = newWidth;
145                     str += word;
146                 }
147 
148                 if(j == wlen - 1){
149                     drawLines.push({text:str, y:height});
150                     // me._drawTextLine(context, str, height);
151                     if(str !== word && width < newWidth) width = newWidth;
152                     height += lineHeight;
153                 }
154             }
155         }
156 
157         me.textWidth = width;
158         me.textHeight = height;
159         if(!me.width) me.width = width;
160         if(!me.height) me.height = height;
161 
162         //vertical alignment
163         var startY = 0;
164         switch(me.textVAlign){
165             case 'middle':
166                 startY = me.height - me.textHeight >> 1;
167                 break;
168             case 'bottom':
169                 startY = me.height - me.textHeight;
170                 break;
171         }
172 
173         //draw background
174         var bg = me.background;
175         if(bg){
176             context.fillStyle = bg;
177             context.fillRect(0, 0, me.width, me.height);
178         }
179 
180         if(me.outline) context.strokeStyle = me.color;
181         else context.fillStyle = me.color;
182 
183         //draw text lines
184         for(var i = 0; i < drawLines.length; i++){
185             var line = drawLines[i];
186             me._drawTextLine(context, line.text, startY + line.y);
187         }
188     },
189 
190     /**
191      * 在指定的渲染上下文上绘制一行文本。
192      * @private
193      */
194     _drawTextLine: function(context, text, y){
195         var me = this, x = 0, width = me.width;
196 
197         switch(me.textAlign){
198             case 'center':
199                 x = width >> 1;
200                 break;
201             case 'right':
202             case 'end':
203                 x = width;
204                 break;
205         };
206 
207         if(me.outline) context.strokeText(text, x, y);
208         else context.fillText(text, x, y);
209     },
210 
211     Statics: /** @lends Text */{
212         /**
213          * 测算指定字体样式的行高。
214          * @param {String} font 指定要测算的字体样式。
215          * @return {Number} 返回指定字体的行高。
216          */
217         measureFontHeight: function(font){
218             var docElement = document.documentElement, fontHeight;
219             var elem = Hilo.createElement('div', {style:{font:font, position:'absolute'}, innerHTML:'M'});
220 
221             docElement.appendChild(elem);
222             fontHeight = elem.offsetHeight;
223             docElement.removeChild(elem);
224             return fontHeight;
225         }
226     }
227 
228 });
