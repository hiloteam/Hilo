  7 /**
  8  * 使用示例:
  9  * <pre>
 10  * var audio = WebSound.getAudio({
 11  *     src: 'test.mp3',
 12  *     loop: false,
 13  *     volume: 1
 14  * }).on('load', function(e){
 15  *     console.log('load');
 16  * }).on('end', function(e){
 17  *     console.log('end');
 18  * }).play();
 19  * </pre>
 20  * @class 声音播放管理器。
 21  * @static
 22  * @module hilo/media/WebSound
 23  * @requires hilo/core/Hilo
 24  * @requires hilo/media/HTMLAudio
 25  * @requires hilo/media/WebAudio
 26  */
 27 var WebSound = {
 28     _audios: {},
 29 
 30     /**
 31      * 激活音频功能。注意：需用户事件触发此方法才有效。目前仅对WebAudio有效。
 32      */
 33     enableAudio: function(){
 34         if(WebAudio.isSupported){
 35             WebAudio.enable();
 36         }
 37     },
 38 
 39     /**
 40      * 获取音频对象。优先使用WebAudio。
 41      * @param {String|Object} source 若source为String，则为音频src地址；若为Object，则需包含src属性。
 42      * @returns {WebAudio|HTMLAudio} 音频播放对象实例。
 43      */
 44     getAudio: function(source){
 45         source = this._normalizeSource(source);
 46         var audio = this._audios[source.src];
 47         if(!audio){
 48             if(WebAudio.isSupported){
 49                 audio = new WebAudio(source);
 50             }else if(HTMLAudio.isSupported){
 51                 audio = new HTMLAudio(source);
 52             }
 53             this._audios[source.src] = audio;
 54         }
 55 
 56         return audio;
 57     },
 58 
 59     /**
 60      * 删除音频对象。
 61      * @param {String|Object} source 若source为String，则为音频src地址；若为Object，则需包含src属性。
 62      */
 63     removeAudio: function(source){
 64         var src = typeof source === 'string' ? source : source.src;
 65         var audio = this._audios[src];
 66         if(audio){
 67             audio.stop();
 68             audio.off();
 69             this._audios[src] = null;
 70             delete this._audios[src];
 71         }
 72     },
 73 
 74     /**
 75      * @private
 76      */
 77     _normalizeSource: function(source){
 78         var result = {};
 79         if(typeof source === 'string') result = {src:source};
 80         else Hilo.copy(result, source);
 81         return result;
 82     }
 83 
 84 };
