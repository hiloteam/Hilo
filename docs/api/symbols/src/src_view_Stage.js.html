  7 /**
  8  * 示例:
  9  * <pre>
 10  * var stage = new Hilo.Stage({
 11  *     container: containerElement,
 12  *     width: 320,
 13  *     height: 480
 14  * });
 15  * </pre>
 16  * @class 舞台是可视对象树的根，可视对象只有添加到舞台或其子对象后才会被渲染出来。创建一个hilo应用一般都是从创建一个stage开始的。
 17  * @augments Container
 18  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。主要有：
 19  * <ul>
 20  * <li><b>container</b>:String|HTMLElement - 指定舞台在页面中的父容器元素。它是一个dom容器或id。若不传入此参数且canvas未被加入到dom树，则需要在舞台创建后手动把舞台画布加入到dom树中，否则舞台不会被渲染。可选。</li>
 21  * <li><b>renderType</b>:String - 指定渲染方式，canvas|dom|webgl，默认canvas。可选。</li>
 22  * <li><b>canvas</b>:String|HTMLCanvasElement|HTMLElement - 指定舞台所对应的画布元素。它是一个canvas或普通的div，也可以传入元素的id。若为canvas，则使用canvas来渲染所有对象，否则使用dom+css来渲染。可选。</li>
 23  * <li><b>width</b>:Number</li> - 指定舞台的宽度。默认为canvas的宽度。可选。
 24  * <li><b>height</b>:Number</li> - 指定舞台的高度。默认为canvas的高度。可选。
 25  * <li><b>paused</b>:Boolean</li> - 指定舞台是否停止渲染。默认为false。可选。
 26  * </ul>
 27  * @module hilo/view/Stage
 28  * @requires hilo/core/Hilo
 29  * @requires hilo/core/Class
 30  * @requires hilo/view/Container
 31  * @requires hilo/renderer/CanvasRenderer
 32  * @requires hilo/renderer/DOMRenderer
 33  * @requires hilo/renderer/WebGLRenderer
 34  * @property {HTMLCanvasElement|HTMLElement} canvas 舞台所对应的画布。它可以是一个canvas或一个普通的div。只读属性。
 35  * @property {Renderer} renderer 舞台渲染器。只读属性。
 36  * @property {Boolean} paused 指示舞台是否暂停刷新渲染。
 37  * @property {Object} viewport 舞台内容在页面中的渲染区域。包含的属性有：left、top、width、height。只读属性。
 38  */
 39 var Stage = Class.create(/** @lends Stage.prototype */{
 40     Extends: Container,
 41     constructor: function(properties){
 42         properties = properties || {};
 43         this.id = this.id || properties.id || Hilo.getUid('Stage');
 44         Stage.superclass.constructor.call(this, properties);
 45 
 46         this._initRenderer(properties);
 47 
 48         //init size
 49         var width = this.width, height = this.height,
 50             viewport = this.updateViewport();
 51         if(!properties.width) width = (viewport && viewport.width) || 320;
 52         if(!properties.height) height = (viewport && viewport.height) || 480;
 53         this.resize(width, height, true);
 54     },
 55 
 56     canvas: null,
 57     renderer: null,
 58     paused: false,
 59     viewport: null,
 60 
 61     /**
 62      * @private
 63      */
 64     _initRenderer: function(properties){
 65         var canvas = properties.canvas;
 66         var container = properties.container;
 67         var renderType = properties.renderType||'canvas';
 68 
 69         if(typeof canvas === 'string') canvas = Hilo.getElement(canvas);
 70         if(typeof container === 'string') container = Hilo.getElement(container);
 71 
 72         if(!canvas){
 73             var canvasTagName = renderType === 'dom'?'div':'canvas';
 74             canvas = Hilo.createElement(canvasTagName, {
 75                 style: {
 76                     position: 'absolute'
 77                 }
 78             });
 79         }
 80         else if(!canvas.getContext){
 81             renderType = 'dom';
 82         }
 83 
 84         this.canvas = canvas;
 85         if(container) container.appendChild(canvas);
 86 
 87         var props = {canvas:canvas, stage:this};
 88         switch(renderType){
 89             case 'dom':
 90                 this.renderer = new DOMRenderer(props);
 91                 break;
 92             case 'webgl':
 93                 if(WebGLRenderer.isSupport()){
 94                     this.renderer = new WebGLRenderer(props);
 95                 }
 96                 else{
 97                     this.renderer = new CanvasRenderer(props);
 98                 }
 99                 break;
100             case 'canvas':
101             default:
102                 this.renderer = new CanvasRenderer(props);
103                 break;
104         }
105     },
106 
107     /**
108      * 添加舞台画布到DOM容器中。注意：此方法覆盖了View.addTo方法。
109      * @param {HTMLElement} domElement 一个dom元素。
110      * @returns {Stage} 舞台本身，可用于链式调用。
111      */
112     addTo: function(domElement){
113         var canvas = this.canvas;
114         if(canvas.parentNode !== domElement){
115             domElement.appendChild(canvas);
116         }
117         return this;
118     },
119 
120     /**
121      * 调用tick会触发舞台的更新和渲染。开发者一般无需使用此方法。
122      * @param {Number} delta 调度器当前调度与上次调度tick之间的时间差。
123      */
124     tick: function(delta){
125         if(!this.paused){
126             this._render(this.renderer, delta);
127         }
128     },
129 
130     /**
131      * 开启/关闭舞台的DOM事件响应。要让舞台上的可视对象响应用户交互，必须先使用此方法开启舞台的相应事件的响应。
132      * @param {String|Array} type 要开启/关闭的事件名称或数组。
133      * @param {Boolean} enabled 指定开启还是关闭。如果不传此参数，则默认为开启。
134      * @returns {Stage} 舞台本身。链式调用支持。
135      */
136     enableDOMEvent: function(type, enabled){
137         var me = this,
138             canvas = me.canvas,
139             types = typeof type === 'string' ? [type] : type,
140             enabled = enabled !== false,
141             handler = me._domListener || (me._domListener = function(e){me._onDOMEvent(e)});
142 
143         for(var i = 0; i < types.length; i++){
144             var type = types[i];
145 
146             if(enabled){
147                 canvas.addEventListener(type, handler, false);
148             }else{
149                 canvas.removeEventListener(type, handler);
150             }
151         }
152 
153         return me;
154     },
155 
156     /**
157      * DOM事件处理函数。此方法会把事件调度到事件的坐标点所对应的可视对象。
158      * @private
159      */
160     _onDOMEvent: function(e){
161         var type = e.type, event = e, isTouch = type.indexOf('touch') == 0;
162 
163         //calculate stageX/stageY
164         var posObj = e;
165         if(isTouch){
166             var touches = e.touches, changedTouches = e.changedTouches;
167             posObj = (touches && touches.length) ? touches[0] :
168                      (changedTouches && changedTouches.length) ? changedTouches[0] : null;
169         }
170 
171         var x = posObj.pageX || posObj.clientX, y = posObj.pageY || posObj.clientY,
172             viewport = this.viewport || this.updateViewport();
173 
174         event.stageX = x = (x - viewport.left) / this.scaleX;
175         event.stageY = y = (y - viewport.top) / this.scaleY;
176 
177         //鼠标事件需要阻止冒泡方法
178         event.stopPropagation = function(){
179             this._stopPropagationed = true;
180         };
181 
182         var obj = this.getViewAtPoint(x, y, true, false, true)||this,
183             canvas = this.canvas, target = this._eventTarget;
184 
185         //fire mouseout/touchout event for last event target
186         var leave = type === 'mouseout';
187         //当obj和target不同 且obj不是target的子元素时才触发out事件
188         if(target && (target != obj && (!target.contains || !target.contains(obj))|| leave)){
189             var out = (type === 'touchmove') ? 'touchout' :
190                       (type === 'mousemove' || leave || !obj) ? 'mouseout' : null;
191             if(out) {
192                 var outEvent = Hilo.copy({}, event);
193                 outEvent.type = out;
194                 outEvent.eventTarget = target;
195                 target._fireMouseEvent(outEvent);
196             }
197             event.lastEventTarget = target;
198             this._eventTarget = null;
199         }
200 
201         //fire event for current view
202         if(obj && obj.pointerEnabled && type !== 'mouseout'){
203             event.eventTarget = this._eventTarget = obj;
204             obj._fireMouseEvent(event);
205         }
206 
207         //set cursor for current view
208         if(!isTouch){
209             var cursor = (obj && obj.pointerEnabled && obj.useHandCursor) ? 'pointer' : '';
210             canvas.style.cursor = cursor;
211         }
212 
213         //fix android: `touchmove` fires only once
214         if(Hilo.browser.android && type === 'touchmove'){
215             e.preventDefault();
216         }
217     },
218 
219     /**
220      * 更新舞台在页面中的可视区域，即渲染区域。当舞台canvas的样式border、margin、padding等属性更改后，需要调用此方法更新舞台渲染区域。
221      * @returns {Object} 舞台的可视区域。即viewport属性。
222      */
223     updateViewport: function(){
224         var canvas = this.canvas, viewport = null;
225         if(canvas.parentNode){
226             viewport = this.viewport = Hilo.getElementRect(canvas);
227         }
228         return viewport;
229     },
230 
231     /**
232      * 改变舞台的大小。
233      * @param {Number} width 指定舞台新的宽度。
234      * @param {Number} height 指定舞台新的高度。
235      * @param {Boolean} forceResize 指定是否强制改变舞台大小，即不管舞台大小是否相同，仍然强制执行改变动作，可确保舞台、画布以及视窗之间的尺寸同步。
236      */
237     resize: function(width, height, forceResize){
238         if(forceResize || this.width !== width || this.height !== height){
239             this.width = width;
240             this.height = height;
241             this.renderer.resize(width, height);
242             this.updateViewport();
243         }
244     }
245 
246 });
