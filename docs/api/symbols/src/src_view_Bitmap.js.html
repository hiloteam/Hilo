  7 /**
  8  * 使用示例:
  9  * <pre>
 10  * var bmp = new Hilo.Bitmap({image:imgElem, rect:[0, 0, 100, 100]});
 11  * stage.addChild(bmp);
 12  * </pre>
 13  * @class Bitmap类表示位图图像类。
 14  * @augments View
 15  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。此外还包括：
 16  * <ul>
 17  * <li><b>image</b> - 位图所在的图像image。必需。</li>
 18  * <li><b>rect</b> - 位图在图像image中矩形区域。</li>
 19  * </ul>
 20  * @module hilo/view/Bitmap
 21  * @requires hilo/core/Hilo
 22  * @requires hilo/core/Class
 23  * @requires hilo/view/View
 24  * @requires hilo/view/Drawable
 25  */
 26  var Bitmap = Class.create(/** @lends Bitmap.prototype */{
 27     Extends: View,
 28     constructor: function(properties){
 29         properties = properties || {};
 30         this.id = this.id || properties.id || Hilo.getUid("Bitmap");
 31         Bitmap.superclass.constructor.call(this, properties);
 32 
 33         this.drawable = new Drawable(properties);
 34 
 35         //init width and height
 36         if(!this.width || !this.height){
 37             var rect = this.drawable.rect;
 38             if(rect){
 39                 this.width = rect[2];
 40                 this.height = rect[3];
 41             }
 42         }
 43     },
 44 
 45     /**
 46      * 设置位图的图片。
 47      * @param {Image|String} image 图片对象或地址。
 48      * @param {Array} rect 指定位图在图片image的矩形区域。
 49      * @returns {Bitmap} 位图本身。
 50      */
 51     setImage: function(image, rect){
 52         this.drawable.init({image:image, rect:rect});
 53         if(rect){
 54             this.width = rect[2];
 55             this.height = rect[3];
 56         }
 57         return this;
 58     }
 59  });
