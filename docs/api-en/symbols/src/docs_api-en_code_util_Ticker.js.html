  7 /**
  8  * @language=en
  9  * @class Ticker is a Timer. It can run the code at specified framerate.
 10  * @param {Number} fps The fps of ticker.
 11  * @module hilo/util/Ticker
 12  * @requires hilo/core/Class
 13  * @requires hilo/core/Hilo
 14  */
 15 var Ticker = Class.create(/** @lends Ticker.prototype */{
 16     constructor: function(fps){
 17         this._targetFPS = fps || 30;
 18         this._interval = 1000 / this._targetFPS;
 19         this._tickers = [];
 20     },
 21 
 22     _paused: false,
 23     _targetFPS: 0,
 24     _interval: 0,
 25     _intervalId: null,
 26     _tickers: null,
 27     _lastTime: 0,
 28     _tickCount: 0,
 29     _tickTime: 0,
 30     _measuredFPS: 0,
 31 
 32     /**
 33      * @language=en
 34      * Start the ticker.
 35      * @param {Boolean} userRAF Whether or not use requestAnimationFrame, default is not.
 36      */
 37     start: function(useRAF){
 38         if(this._intervalId) return;
 39         this._lastTime = +new Date();
 40 
 41         var self = this, interval = this._interval,
 42             raf = window.requestAnimationFrame ||
 43                   window[Hilo.browser.jsVendor + 'RequestAnimationFrame'];
 44 
 45         if(useRAF && raf){
 46             var tick = function(){
 47                 self._tick();
 48             }
 49             var runLoop = function(){
 50                 self._intervalId = setTimeout(runLoop, interval);
 51                 raf(tick);
 52             };
 53         }else{
 54             runLoop = function(){
 55                 self._intervalId = setTimeout(runLoop, interval);
 56                 self._tick();
 57             };
 58         }
 59 
 60         runLoop();
 61     },
 62 
 63     /**
 64      * @language=en
 65      * Stop the ticker.
 66      */
 67     stop: function(){
 68         clearTimeout(this._intervalId);
 69         this._intervalId = null;
 70         this._lastTime = 0;
 71     },
 72 
 73     /**
 74      * @language=en
 75      * Pause the ticker.
 76      */
 77     pause: function(){
 78         this._paused = true;
 79     },
 80 
 81     /**
 82      * @language=en
 83      * Resume the ticker.
 84      */
 85     resume: function(){
 86         this._paused = false;
 87     },
 88 
 89     /**
 90      * @private
 91      */
 92     _tick: function(){
 93         if(this._paused) return;
 94         var startTime = +new Date(),
 95             deltaTime = startTime - this._lastTime,
 96             tickers = this._tickers;
 97 
 98         //calculates the real fps
 99         if(++this._tickCount >= this._targetFPS){
100             this._measuredFPS = 1000 / (this._tickTime / this._tickCount) + 0.5 >> 0;
101             this._tickCount = 0;
102             this._tickTime = 0;
103         }else{
104             this._tickTime += startTime - this._lastTime;
105         }
106         this._lastTime = startTime;
107 
108         var tickersCopy = tickers.slice(0);
109         for(var i = 0, len = tickersCopy.length; i < len; i++){
110             tickersCopy[i].tick(deltaTime);
111         }
112     },
113 
114     /**
115      * @language=en
116      * Get the fps.
117      */
118     getMeasuredFPS: function(){
119         return this._measuredFPS;
120     },
121 
122     /**
123      * @language=en
124      * Add tickObject. The tickObject must implement the tick method.
125      * @param {Object} tickObject The tickObject to add.It must implement the tick method.
126      */
127     addTick: function(tickObject){
128         if(!tickObject || typeof(tickObject.tick) != 'function'){
129             throw new Error('Ticker: The tick object must implement the tick method.');
130         }
131         this._tickers.push(tickObject);
132     },
133 
134     /**
135      * @language=en
136      * Remove the tickObject
137      * @param {Object} tickObject The tickObject to remove.
138      */
139     removeTick: function(tickObject){
140         var tickers = this._tickers,
141             index = tickers.indexOf(tickObject);
142         if(index >= 0){
143             tickers.splice(index, 1);
144         }
145     },
146     /**
147      * 下次tick时回调
148      * @param  {Function} callback
149      * @return {tickObj}
150      */
151     nextTick:function(callback){
152         var that = this;
153         var tickObj = {
154             tick:function(dt){
155                 that.removeTick(tickObj);
156                 callback();
157             }
158         };
159 
160         that.addTick(tickObj);
161         return tickObj;
162     },
163     /**
164      * 延迟指定的时间后调用回调, 类似setTimeout
165      * @param  {Function} callback
166      * @param  {Number}   duration 延迟的毫秒数
167      * @return {tickObj}
168      */
169     timeout:function(callback, duration){
170         var that = this;
171         var targetTime = new Date().getTime() + duration;
172         var tickObj = {
173             tick:function(){
174                 var nowTime = new Date().getTime();
175                 var dt = nowTime - targetTime;
176                 if(dt >= 0){
177                     that.removeTick(tickObj);
178                     callback();
179                 }
180             }
181         };
182         that.addTick(tickObj);
183         return tickObj;
184     },
185     /**
186      * 指定的时间周期来调用函数, 类似setInterval
187      * @param  {Function} callback
188      * @param  {Number}   duration 时间周期，单位毫秒
189      * @return {tickObj}
190      */
191     interval:function(callback, duration){
192         var that = this;
193         var targetTime = new Date().getTime() + duration;
194         var tickObj = {
195             tick:function(){
196                 var nowTime = new Date().getTime();
197                 var dt = nowTime - targetTime;
198                 if(dt >= 0){
199                     if(dt < duration){
200                         nowTime -= dt;
201                     }
202                     targetTime = nowTime + duration;
203                     callback();
204                 }
205             }
206         };
207         that.addTick(tickObj);
208         return tickObj;
209     }
210 });
